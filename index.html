function savePoster() { const poster = document.getElementById('poster-canvas'); const btn = document.querySelector('.btn-save'); btn.innerText = "CAPTURING HD..."; // Use html2canvas with high-performance settings html2canvas(poster, { scale: 4, // Forces 4K-level detail useCORS: true, logging: false, backgroundColor: null, // Maintains transparency if any imageTimeout: 0, onclone: (clonedDoc) => { // Ensures the image in the clone maintains its quality const clonedImg = clonedDoc.getElementById('bg-image'); clonedImg.style.imageRendering = 'crisp-edges'; } }).then(canvas => { // Force highest quality PNG format const link = document.createElement('a'); link.download = 'Austine_Pro_Design_HD.png'; link.href = canvas.toDataURL("image/png", 1.0); // 1.0 = 100% quality link.click(); btn.innerText = "EXPORT IMAGE"; }); }
